7   package tetris.logiikka;
8   
9   import java.util.HashMap;
10  import java.util.Random;
11  import java.util.function.Supplier;
12  import tetris.palat.IPala;
13  import tetris.palat.JPala;
14  import tetris.palat.LPala;
15  import tetris.palat.OPala;
16  import tetris.palat.Ruutu;
17  import tetris.palat.SPala;
18  import tetris.palat.TPala;
19  import tetris.palat.ZPala;
20  
21  /**
22   * PalikanArpoja arpoo palan
23   * @author samukaup
24   */
25  public class PalikanArpoja {
26      private Random random;
27      private HashMap<Integer, Supplier<Ruutu>> palat;
28  
29      public PalikanArpoja() {
30          random = new Random();
31          palat = new HashMap<>();
32          palat.put(0, () -> new TPala(0, 0));
33          palat.put(1, () -> new SPala(0, 0));
34          palat.put(2, () -> new ZPala(0, 0));
35          palat.put(3, () -> new OPala(0, 0));
36          palat.put(4, () -> new IPala(0, 0));
37          palat.put(5, () -> new JPala(0, 0));
38          palat.put(6, () -> new LPala(0, 0));
39      }
40      
41      /**
42       * 
43       * @return 
44       */
45      
46      public Ruutu arvoPala() {
47          int luku = random.nextInt(6);
48          return haePala(luku);
49          
50      }
51      
52      /**
53       * 
54       * @param x
55       * @return 
56       */
57      
58      public Ruutu haePala(int x) {
59          return palat.get(x).get();
60      }
61      
62  }
