1   package tetris.gui;
2   
3   import java.awt.Color;
4   import java.awt.Font;
5   import java.awt.Graphics;
6   import java.util.List;
7   import java.util.Random;
8   import javax.swing.JPanel;
9   import tetris.blocks.Block;
10  import tetris.logic.Board;
11  import tetris.blocks.Shape;
12  import tetris.logic.Tetris;
13  
14  /**
15   * Paneeli luokka, joka piirtää pelin komponentit.
16   *
17   * @author Rubiini
18   */
19  public class Panel extends JPanel implements Update {
20  
21      private Tetris tetris;
22      private int widthSize;
23      private int heightSize;
24  
25      /**
26       * Panel asettaa paikallisille muuttujille konstruktorille annetut arvot.
27       * @param tetris Tetris pelin logiikka
28       * @param heightSize Koko pysty-suunnassa
29       * @param widthSize Koko vaaka-suunnassa
30       */
31      public Panel(Tetris tetris, int heightSize, int widthSize) {
32          this.tetris = tetris;
33          this.widthSize = widthSize;
34          this.heightSize = heightSize;
35      }
36  
37      @Override
38      public void paintComponent(Graphics g) {
39          super.paintComponent(g);
40          
41          g.setColor(Color.BLACK);
42          
43          paintShape(g);
44          paintBoard(g);
45          paintScore(g);
46      }
47      
48      /**
49       * Piirtää liikutettavan palan.
50       * @param g 
51       */
52      public void paintShape(Graphics g) {
53          List<Block> shape = tetris.getShape().getList();
54          for (Block block : shape) {
55              g.fill3DRect(block.getX() * widthSize, block.getY() * heightSize, widthSize, heightSize, true);
56          }
57      }
58      
59      /**
60       * Piirtää pelialustalla olevat palat.
61       * @param g 
62       */
63      public void paintBoard(Graphics g) {
64          int[][] matrix = tetris.getBoard().getBoardMatrix();
65          for (int i = 0; i < 20; i++) {
66              for (int j = 0; j < 10; j++) {
67                  if (matrix[i][j] == 1) {
68                      g.fill3DRect(j * widthSize, i * heightSize, widthSize, heightSize, true);
69                  } 
70              }
71          }
72      }
73      /**
74       * Piirtää pistenäkymän.
75       * @param g 
76       */
77      public void paintScore(Graphics g) {
78          g.setFont(new Font("Arial", Font.PLAIN, 16));
79          g.drawString("Score: " + tetris.getScore(), 
80                  tetris.getHeight() - g.getFontMetrics().stringWidth("Score: " + tetris.getScore()) + 270, 
81                  g.getFontMetrics().getHeight());
82      }
83      /**
84       * Päivittää käyttöliittymän näkymän, eli maalaa palat uudestaan.
85       */
86      
87      @Override
88      public void update() {
89          super.repaint();
90      }
91  
92  }
